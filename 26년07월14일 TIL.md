struct RecipeNode
{

    Ingredient matter;
    shared_ptr<RecipeNode> nextMatter = nullptr;

};

    기존 과제에서는 재료가 2개로 고정되어 있었지만, 
    연결 리스트(Linked List) 구조를 사용하여 구현하였다.
    이를 통해 재료 개수가 고정되지 않도록 설계하였으며, 향후 재료가 추가되더라도 구조를 수정하지 않고 확장할 수 있도록 구현하였다.


class Potion
{

protected:

    string name;
    shared_ptr<RecipeNode> head = nullptr;

public:

    Potion(string name, shared_ptr<RecipeNode> recipe);

//function
    void PrintRecipe() const;

//Setter
    void setName(const string name);
    void setRecipe(shared_ptr<RecipeNode> recipe);

//Getter
    string getName() const;
    shared_ptr<RecipeNode> getRecipe() const;

}; 


    RecipeNode는 shared_ptr를 사용하여 연결하였으며, 참조가 모두 사라질 경우 자동으로 메모리가 해제되도록 구현하였다. 이를 통해 수동 delete를 사용하지 않아도 되며 메모리 관리의 부담을 줄일 수 있었다.
    

struct Ingredient
{

    string name;
    int count;

};

struct RecipeNode
{

    Ingredient matter;
    shared_ptr<RecipeNode> nextMatter = nullptr;

};


class Potion
{

protected:

    string name;
    shared_ptr<RecipeNode> head = nullptr;

public:

    Potion(string name, shared_ptr<RecipeNode> recipe);

//function
    void PrintRecipe() const;

//Setter
    void setName(const string name);
    void setRecipe(shared_ptr<RecipeNode> recipe);

//Getter
    string getName() const;
    shared_ptr<RecipeNode> getRecipe() const;

}; 


    마지막으로 매우 특이한 경험을 하였는데, C++에서는 클래스를 정의하는 과정에서 사용하는 타입이 먼저 선언되어 있어야 한다. 따라서 클래스 내부에서 구조체를 멤버 변수로 사용할 경우, 해당 구조체는 클래스보다 먼저 선언되어 있어야 한다는 점을 다시 확인하였다.
