- UpgradeSetup.cpp : 종속성과 의도된 재사용 관계의 경계에 대한 고찰을 진행할 수 있었던 코드이다.
처음에는 PrintStatus()를 재사용하면 UpgradeSetup.cpp가 CharacterSetup.cpp의 함수에 종속성을 갖게 된다고 판단했다.
하지만 PrintStatus()는 CharacterSetup의 핵심 로직이 아니라 캐릭터 상태를 출력하는 역할만 담당하는 함수이며, 내부 상태를 변경하거나 복잡한 로직을 수행하지 않는다. 따라서 해당 기능의 재사용은 높은 결합을 발생시키는 위험한 의존 관계라기보다는, 출력 형식을 공유하기 위한 허용 가능한 수준의 의존 관계라고 판단하였다.
결과적으로 동일한 cout 구문을 여러 번 작성하는 것보다 PrintStatus()를 재사용하여 출력 형식을 일관되게 유지하고, 코드 중복으로 발생할 수 있는 휴먼 에러 가능성을 낮추는 것이 더 적절하다고 판단했다.


#include "UpgradeSetup.h"

bool UpgradeSetup(int InsertStat[])
{
    int choice;
    int HpUpPotion = 5;
    int MpUpPotion = 5;
    bool isGameStart = false;

    while (!isGameStart)
    {

        cout << "You have " << HpUpPotion << " HP Up potions left" << endl;
        cout << "You have " << MpUpPotion << " MP Up potions left" << endl;
        cout << "<You have a chance to upgrade your stats!>" << endl;
        cout << "====================================================" << endl;
        cout << "1.Use HP Up 20" << "2.Use MP Up 20" << "3.Attack x 2" << endl;
        cout << "4.Defense x 2" << "5.Show Stats" << "0.Start Game" << endl;
        cout << "Please choose an option (0-5): ";
        cout << "====================================================" << endl;
        cin >> choice;

        switch(choice)
        {

            case 0:
                isGameStart = true;
                cout << "Starting the game..." << endl;
                break;
            case 1:
                if (HpUpPotion > 0)
                {
                    InsertStat[0] += 20;
                    HpUpPotion--;
                    cout << "HP +20! Current HP: " << InsertStat[0] << endl;
                    cout << "You have " << HpUpPotion << " HP Up potions left." << endl;
                }
                else
                {
                    cout << "You don't have any HP Up potions left!" << endl;
                }
                break;
            case 2:
                if (MpUpPotion > 0)
                {
                    InsertStat[1] += 20;
                    MpUpPotion--;
                    cout << "MP +20! Current MP: " << InsertStat[1] << endl;
                    cout << "You have " << MpUpPotion << " MP Up potions left." << endl;
                }
                else
                {
                    cout << "You don't have any MP Up potions left!" << endl;
                }
                break;
            case 3:
                InsertStat[2] *= 2;
                break;
            case 4:
                InsertStat[3] *= 2;
                break;
            case 5:
                PrintStatus("Your character", InsertStat);
                break;
            default:
                cout << "Invalid choice. No stats were upgraded." << endl;
                break;
        }
    }

    return isGameStart;

}
